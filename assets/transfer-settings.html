const http = require('http');

const server = http.createServer((req, res) => {
  if (req.url === '/api/incoming-payments/receive' && req.method === 'POST') {
    let body = '';
    req.on('data', chunk => { body += chunk; });
    req.on('end', () => {
      const data = JSON.parse(body);

      console.log('[SMS Gateway] New transaction:');
      console.log('  Type:', data.type);
      console.log('  Provider:', data.provider);
      console.log('  Amount:', data.amount);
      console.log('  Sender:', data.sender_number);
      console.log('  Receiver:', data.receiver_number);
      console.log('  Fee:', data.transfer_fee);
      console.log('  Balance:', data.balance_after);
      console.log('  Message:', data.message_text);

      res.writeHead(200, { 'Content-Type': 'application/json' });
      res.end(JSON.stringify({ success: true }));
    });
  } else {
    res.writeHead(404);
    res.end();
  }
});

server.listen(3000, () => {
  console.log('Webhook server running on port 3000');
});
